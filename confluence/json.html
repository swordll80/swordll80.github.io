1 JSON 定义 RFC 4627
// JSON = JavaScript Object Notation // 轻量级数据交换格式, 语言无关 // 基于两个结构: 键值对集合 + 有序值列表
2 基本示例 RFC 4627
{ "name": "Alice", "age": 25, "isStudent": false, "hobbies": ["reading", "coding"], "address": { "city": "Beijing", "zip": "100000" } }
3 JSON 数据类型 RFC 4627
// 1. string → "hello" // 2. number → 42, 3.14, -10, 1.5e10 // 3. boolean → true / false // 4. null → null // 5. object → { "key": "value" } // 6. array → [1, 2, 3]
4 字符串规则 RFC 4627
// 必须用双引号 (不能用单引号) // 转义: \" \\ \/ \b \f \n \r \t \uXXXX { "message": "He said \"Hello\"\nNew line" }
5 数字规则 RFC 4627
// 整数: 无前导零 (不能 0123) // 小数: 必须有数字在小数点前后 // 指数: e 或 E, 后可跟 + 或 - // 不支持: NaN, Infinity, undefined // 不支持: 八进制, 十六进制
6 JSON 格式规则 RFC 7159
// 1. 最外层可以是 {} 或 [] (RFC 7159+) // 2. 键必须是字符串 (双引号) // 3. 不能有尾随逗号 // 4. 不支持注释 // 5. 空白(空格/制表/换行)被忽略
7 Schema 基础 JSON Schema
{ "$schema": "https://json-schema.org/draft/2020-12/schema", "$id": "https://example.com/person.schema.json", "title": "Person", "type": "object", "properties": { "name": { "type": "string" }, "age": { "type": "integer", "minimum": 0 } }, "required": ["name"] }
8 类型约束 JSON Schema
// "type": "string" / "number" / "integer" // "type": "object" / "array" / "boolean" / "null" // 可多类型: "type": ["string", "null"] // 字符串: "minLength", "maxLength", "pattern" (正则) // 数值: "minimum", "maximum", "multipleOf" // 数组: "minItems", "maxItems", "uniqueItems" // 对象: "minProperties", "maxProperties"
9 枚举与常量 JSON Schema
{ "color": { "enum": ["red", "green", "blue"] }, "version": { "const": "1.0" } }
10 组合约束 JSON Schema
{ "allOf": [{...}, {...}], // 全部满足 "anyOf": [{...}, {...}], // 任一满足 "oneOf": [{...}, {...}], // 恰好一个满足 "not": {...} // 不满足 }
11 引用 ($ref) JSON Schema
{ "definitions": { "address": { "type": "object", "properties": {...} } }, "properties": { "home": { "$ref": "#/definitions/address" }, "work": { "$ref": "#/definitions/address" } } }
12 JSONPath 表达式 JSONPath
// $ → 根元素 // $.store.book[0] → 第一本书 // $.store.book[*] → 所有书 // $..author → 所有 author (递归) // $.store.book[?(@.price < 10)] → 筛选 // $.store.book[0,1] → 前两本 // $.store.book[-1] → 最后一本 // $.store.book[:2] → 前两本(切片) // $.store.book[?(@.isbn)] → 有 isbn 的书
13 JSON.stringify ES5
const obj = { name: "Alice", age: 25 }; const json = JSON.stringify(obj); // '{"name":"Alice","age":25}' const pretty = JSON.stringify(obj, null, 2); // 第二个参数: 过滤/转换函数或属性数组 // 第三个参数: 缩进空格数
14 JSON.parse ES5
const obj = JSON.parse('{"name":"Alice","age":25}'); const obj2 = JSON.parse(json, (key, value) => { // 第二个参数: reviver 函数, 转换解析后的值 if (key === 'date') return new Date(value); return value; });
15 JSON 不支持的类型 ES5
// JSON.stringify 会忽略/转换以下值: // undefined → 忽略 (对象属性) 或 null (数组) // Symbol → 忽略 // Function → 忽略 // NaN / Infinity → null // Date → .toISOString() 字符串 // 循环引用 → 抛 TypeError
16 toJSON 方法 ES5
const obj = { date: new Date(), toJSON() { return { custom: 'format' }; } }; // JSON.stringify 会调用对象的 toJSON()
17 json.dumps / json.loads 3.x
import json s = json.dumps({'name': 'Alice', 'age': 25}, indent=2) data = json.loads(s)
18 json.dump / json.load 3.x
with open('data.json', 'w') as f: json.dump(data, f, indent=2, ensure_ascii=False) with open('data.json', 'r') as f: data = json.load(f)
19 Python-JSON 类型映射 3.x
# dict → object, list/tuple → array # str → string, int/float → number # True/False → true/false, None → null # datetime 等自定义类型需手动转换 class MyEncoder(json.JSONEncoder): def default(self, obj): if isinstance(obj, datetime): return obj.isoformat() return super().default(obj)
20 C++ (nlohmann/json) C++11
// 最流行的 C++ JSON 库 #include <nlohmann/json.hpp> using json = nlohmann::json; json j = json::parse("{\"name\":\"Alice\"}"); std::string name = j["name"]; std::string s = j.dump(2);

25 JSONC (JSON with Comments) JSONC
// VS Code 等编辑器支持的格式 // JSON + // 和 /**/ 注释 // 用于 tsconfig.json, settings.json 等

27 BSON BSON
// Binary JSON, MongoDB 使用的二进制格式 // 支持更多类型: Date, ObjectId, Binary data // 解析更快, 但更占空间
28 MessagePack -
// 二进制序列化格式, 比 JSON 更紧凑 // 兼容 JSON 类型, 但体积更小
29 Web API -
// RESTful API 的标准数据格式 // Content-Type: application/json // 请求体和响应体通常都是 JSON
30 配置文件 -
// package.json (npm), composer.json (PHP) // tsconfig.json, .prettierrc, .eslintrc // 现代项目的主流配置格式
31 数据存储 -
// NoSQL 数据库: MongoDB, CouchDB, Firebase // 浏览器: localStorage, sessionStorage // 移动端: SharedPreferences, NSUserDefaults

// 微服务间通信, 消息队列 (Kafka, RabbitMQ) // WebSocket 消息, RPC 调用 // 日志格式 (结构化日志)
33 尾随逗号 -
// 错误: {"a": 1, "b": 2,} → 解析失败! // JSON 标准不允许尾随逗号 // JavaScript 对象允许, 但 JSON 不允许
34 单引号字符串 -
// 错误: {'key': 'value'} → 解析失败! // JSON 必须使用双引号
35 数字前导零 -
// 错误: {"zip": 01234} → 解析失败! // 正确: {"zip": "01234"} (用字符串)
36 特殊值处理 -
// NaN, Infinity, undefined → 无法序列化 // 需替换为 null 或字符串 // 日期需转为 ISO 8601 字符串
37 循环引用 -
// const a = {}; const b = {a}; a.b = b; // JSON.stringify(a) → TypeError // 解决: 用 toJSON 或 flatted 库
